  if (!DIFY_API_KEY) {

  }

**3. Create `app/api/brief/parameters/route.ts`** — New GET route that proxies to Dify `/parameters` endpoint. Per D-02, this is called on page mount to fetch `opening_statement` and `suggested_questions`:

```typescript
import { NextResponse } from 'next/server';
import { getParameters } from '@/lib/dify/client';

export async function GET() {
  try {
    const response = await getParameters();

    if (!response.ok) {
      return NextResponse.json(
        { error: 'Failed to fetch parameters', code: 'DIFY_PARAMETERS_ERROR', status: response.status },
        { status: response.status }
      );
    }

    const data = await response.json();
    return NextResponse.json({
      opening_statement: data.opening_statement || '',
      suggested_questions: data.suggested_questions || [],
    });
  } catch (err) {
    const errMsg = err instanceof Error ? err.message : '';
    if (errMsg.includes('not configured')) {
      return NextResponse.json(
        { error: 'Chat service is not configured', code: 'DIFY_NOT_CONFIGURED', status: 500 },
        { status: 500 }
      );
    }
    return NextResponse.json(
      { error: 'Unable to reach the chat service', code: 'DIFY_UNREACHABLE', status: 502 },
      { status: 502 }
    );
  }
}
```

**Hook interface (per D-01):**
```typescript
export interface BriefMessage {
  id: string;
  role: 'user' | 'assistant';
  content: string;
}

export interface UseBriefChatReturn {
  messages: BriefMessage[];
  suggestedQuestions: string[];
  input: string;
  setInput: (value: string) => void;
  sendMessage: (text?: string) => void;
  isLoading: boolean;
  isBriefComplete: boolean;
}
```

**Implementation details:**

1. **State:**
   - `messages: BriefMessage[]` — chat history
   - `input: string` — controlled input value
   - `isLoading: boolean` — true while streaming
   - `isBriefComplete: boolean` — true after [BRIEF_COMPLETE] detected
   - `suggestedQuestions: string[]` — from parameters fetch
   - `conversationIdRef: useRef<string | null>(null)` — use ref to avoid stale closure (Pitfall 2 from research)

2. **Initialization (useEffect on mount, per D-02 and D-03):**
   - Fetch `GET /api/brief/parameters`
   - On success: set `opening_statement` as first assistant message (id: `'opening'`, role: `'assistant'`), set `suggestedQuestions` from response
   - On failure: use fallback opening statement `"Hi! Tell me about the email you'd like to create."` and empty suggestions array (Pitfall 5). Log error to console.

3. **sendMessage function:**
   - Accept optional `text` parameter (for chip clicks per D-03) — if provided use it, otherwise use `input` state
   - Guard: return early if `isLoading` is true or message is empty (Pitfall 3)
   - Add user message to `messages` array (id: `crypto.randomUUID()`, role: `'user'`)
   - Clear `input` to empty string
   - Set `isLoading = true`
   - Create assistant placeholder message (id: `crypto.randomUUID()`, role: `'assistant'`, content: `''`)
   - POST to `/api/brief/chat` with `{ message: text, conversation_id: conversationIdRef.current || '' }`
   - Parse SSE stream using ReadableStream reader (same buffer-and-split pattern as the server route):
     ```
     buffer += decoder.decode(value, { stream: true });
     const lines = buffer.split('\n');
     buffer = lines.pop() || '';
     for (const line of lines) {
       if (!line.startsWith('data: ')) continue;
       const data = JSON.parse(line.slice(6));
       // handle data.answer, data.event === 'done', data.event === 'error'
     }
     ```
   - For `answer` chunks: accumulate into a local `accumulated` string, update the last assistant message content with `accumulated`
   - For `conversation_id`: update `conversationIdRef.current` (from first chunk that has it)
   - For `event: 'done'`: check `accumulated` for `[BRIEF_COMPLETE]` marker (Pitfall 4 — ONLY check after done, not during streaming). If found: strip marker with `accumulated.replace('[BRIEF_COMPLETE]', '').trim()`, update message content, set `isBriefComplete = true` per D-10
   - For `event: 'error'`: update assistant message content with `"Sorry, something went wrong. Please try again."`
   - Wrap JSON.parse in try/catch to skip malformed lines
   - Set `isLoading = false` in finally block
   - Clear `suggestedQuestions` to empty array after first user message (per D-04 — only initial suggested questions used)

4. **Message state updates:** Use functional `setMessages` to avoid stale state. When updating the streaming assistant message, replace the last message in the array by matching its id.

5. **Export:** `export function useBriefChat(): UseBriefChatReturn`

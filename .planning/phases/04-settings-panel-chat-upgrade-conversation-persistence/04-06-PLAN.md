1. Rewrite `components/assistant-ui/brief-runtime-provider.tsx` to use `useRemoteThreadListRuntime`:

Replace `useLocalRuntime` with `useRemoteThreadListRuntime`. The adapter pattern from the research:

```typescript
'use client';

import {
  AssistantRuntimeProvider,
  useLocalRuntime,
  useRemoteThreadListRuntime,
  type RemoteThreadListAdapter,
} from '@assistant-ui/react';
import { useMemo, useCallback, useRef } from 'react';
import { createDifyAdapter } from '@/lib/dify/adapter';
import { basePath } from '@/lib/base-path';

interface BriefRuntimeProviderProps {
  children: React.ReactNode;
  onBriefContent?: (content: string) => void;
}

export function BriefRuntimeProvider({ children, onBriefContent }: BriefRuntimeProviderProps) {
  const threadListAdapter: RemoteThreadListAdapter = useMemo(() => ({
    async list() {
      const res = await fetch(`${basePath}/api/threads`);
      const data = await res.json();
      return {
        threads: data.threads.map((t: any) => ({
          remoteId: t.id,
          status: t.is_archived ? ('archived' as const) : ('regular' as const),
          title: t.title ?? undefined,
        })),
      };
    },
    async initialize(localId: string) {
      const res = await fetch(`${basePath}/api/threads`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ localId }),
      });
      const result = await res.json();
      return { remoteId: result.id };
    },
    async rename(remoteId: string, title: string) {
      await fetch(`${basePath}/api/threads/${remoteId}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ title }),
      });
    },
    async archive(remoteId: string) {
      await fetch(`${basePath}/api/threads/${remoteId}/archive`, { method: 'POST' });
    },
    async unarchive(remoteId: string) {
      await fetch(`${basePath}/api/threads/${remoteId}/unarchive`, { method: 'POST' });
    },
    async delete(remoteId: string) {
      await fetch(`${basePath}/api/threads/${remoteId}`, { method: 'DELETE' });
    },
    async fetch(remoteId: string) {
      const res = await fetch(`${basePath}/api/threads/${remoteId}`);
      const t = await res.json();
      return {
        remoteId: t.id,
        status: t.is_archived ? ('archived' as const) : ('regular' as const),
        title: t.title ?? undefined,
      };
    },
    async generateTitle(remoteId: string, messages: any[]) {
      const firstUserMsg = messages.find((m: any) => m.role === 'user');
      const title = firstUserMsg
        ? (firstUserMsg.content?.[0] as any)?.text?.slice(0, 50) || 'New conversation'
        : 'New conversation';

      await fetch(`${basePath}/api/threads/${remoteId}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ title }),
      });

      const { createAssistantStream } = await import('assistant-stream');
      return createAssistantStream(async (controller: any) => {
        controller.appendText(title);
      });
    },
  }), []);

  const runtime = useRemoteThreadListRuntime({
    runtimeHook: useCallback(() => {
      // Create adapter for the current thread
      // The adapter will send threadId along with messages so the API route can persist them
      const adapter = createDifyAdapter({
        onBriefContent,
      });
      return useLocalRuntime(adapter);
    }, [onBriefContent]),
    adapter: threadListAdapter,
  });

  return (
    <AssistantRuntimeProvider runtime={runtime}>
      {children}
    </AssistantRuntimeProvider>
  );
}
```

function DynamicSuggestions() {
  const [prompts, setPrompts] = useState<Array<{ id: string; name: string; text: string; autoSend: boolean }>>([]);

  useEffect(() => {
    fetch(`${basePath}/api/test-prompts`)
      .then(res => res.json())
      .then(data => setPrompts(data.slice(0, 6)))
      .catch(() => {}); // Silently fail -- suggestions are non-critical
  }, []);

  if (prompts.length === 0) return null;

  return (
    <div className="flex flex-wrap gap-2 px-4 py-2">
      {prompts.map(p => (
        <ThreadPrimitive.Suggestion
          key={p.id}
          prompt={p.text}
          method="replace"
          {...(p.autoSend ? { autoSend: true } : {})}
        >
          <span className="px-3 py-1.5 text-xs rounded-full border border-border bg-background hover:bg-muted transition-colors text-foreground cursor-pointer">
            {p.name}
          </span>
        </ThreadPrimitive.Suggestion>
      ))}
    </div>
  );
}

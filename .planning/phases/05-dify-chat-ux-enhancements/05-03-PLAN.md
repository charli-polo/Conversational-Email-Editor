```tsx
function AssistantMessageContent() {
  const message = useMessage();
  const metadata = (message as Record<string, unknown>)?.message?.metadata;
  const custom = (metadata as Record<string, unknown>)?.custom as Record<string, unknown> | undefined;
  const isOpener = custom?.isOpener === true;
  const suggestedQuestions = (custom?.suggestedQuestions as string[]) || [];
  const isStreamingReasoning = custom?.isStreamingReasoning === true;
  const streamingTools = (custom?.streamingTools as string[]) || [];
  
  // Extract reasoning parts from content
  const content = (message as Record<string, unknown>)?.message?.content;
  const reasoningParts = Array.isArray(content)
    ? content.filter((p: Record<string, unknown>) => p.type === 'reasoning').map((p: Record<string, unknown>) => ({ text: p.text as string }))
    : [];

  // Extract reasoning text for streaming expand view
  const reasoningText = reasoningParts.map(r => r.text).join('\n\n');

  // Check if message is still streaming (isRunning)
  // useMessage() provides the message state -- check status
  const isRunning = (message as Record<string, unknown>)?.isRunning === true ||
    ((message as Record<string, unknown>)?.message as Record<string, unknown>)?.status?.type === 'running';

  // Extract image/document attachment parts from content for D-12 (WARNING 2 fix)
  const attachmentParts = Array.isArray(content)
    ? content.filter((p: Record<string, unknown>) => p.type === 'image' || p.type === 'document')
    : [];

  return (
    <>
      <div className="w-full text-sm text-foreground prose prose-sm dark:prose-invert max-w-none prose-p:my-1 prose-ul:my-1 prose-ol:my-1 prose-li:my-0.5 prose-headings:my-2">
        <MessagePrimitive.Content
          components={{
            Text: ({ text }) => <Markdown>{text}</Markdown>,
          }}
        />
      </div>
      
      {/* D-12: Render image/document attachments inline in assistant messages (WARNING 2 fix) */}
      {attachmentParts.length > 0 && (
        <div className="flex flex-wrap gap-2 mt-2">
          {attachmentParts.map((part: Record<string, unknown>, i: number) => (
            <MessageAttachmentDisplay
              key={i}
              type={part.type as 'image' | 'document'}
              name={(part.name as string) || 'attachment'}
              url={part.url as string | undefined}
            />
          ))}
        </div>
      )}
      
      {/* D-14/D-16: Streaming reasoning indicator -- dots + timer while agent is thinking */}
      {isStreamingReasoning && isRunning && (
        <StreamingReasoningIndicator
          tools={streamingTools}
          reasoningText={reasoningText}
        />
      )}
      
      {/* D-17: Post-response collapsible reasoning -- only when streaming is done and reasoning exists */}
      {!isStreamingReasoning && !isRunning && reasoningParts.length > 0 && (
        <ReasoningSection reasoningParts={reasoningParts} />
      )}
      
      {isOpener && suggestedQuestions.length > 0 && <OpenerSuggestions suggestions={suggestedQuestions} />}
      <AssistantActionToolbar />
    </>
  );
}
```

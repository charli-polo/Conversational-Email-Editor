```typescript
const visibleTabs = useMemo(() => {
  const tagMap = new Map<string, ConversationTag>();
  for (const c of conversations) {
    for (const t of c.tags) {
      if (!tagMap.has(t.id)) {
        tagMap.set(t.id, t);
      }
    }
  }
  return Array.from(tagMap.values()).sort((a, b) => a.name.localeCompare(b.name));
}, [conversations]);
```
